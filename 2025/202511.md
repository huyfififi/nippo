```cpp
#include <utility>

class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int unique_index = 0;
        int i = 0;
        while (i < nums.size()) {
            while (i + 1 < nums.size() && nums[i] == nums[i + 1]) {
                ++i;
            }
            std::swap(nums[unique_index], nums[i]);
            ++i;
            ++unique_index;
        }
        return unique_index;
    }
};
```
