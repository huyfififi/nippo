def possible(a: int, b: int) -> bool:
    for apc in range(a + 1):
        for bpc in range(b + 1):
            if apc - (a - apc) + 2 * bpc - 2 * (b - bpc) == 0:
                return True
    return False


for _ in range(int(input())):
    a, b = map(int, input().split())
    if possible(a, b):
        print("YES")
    else:
        print("NO")
