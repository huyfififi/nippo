## Retry: Radix Sort

```python
def radix_sort(arr: list[int]) -> list[int]:
    max_element: int = max(arr)

    i: int = 1
    while i <= max_element:
        bucket = []
        for j in range(10):  # decimal
            bucket.append([])

        for e in arr:
            bucket[e // i % 10].append(e)

        arr: list[int] = []
        for j in range(10):
            arr.extend(bucket[j])

        i *= 10

    return arr


arr: list[int] = [123, 45, 67, 890, 10000, 20, 30, 44, 1, 55]
print(arr)
print(radix_sort(arr))
```
