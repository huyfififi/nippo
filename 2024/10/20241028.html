def count_disturbance(a: list[int]) -> int:
    count: int = 0
    for i in range(len(a) - 1):
        if a[i] == a[i + 1]:
            count += 1
    return count


def minimize_disturbance(a: list[int]) -> int:

    for i in range(n - 1):
        if a[i] == a[i + 1] or a[n - i - 1] == a[n - i - 2]:
            a[i + 1], a[n - i - 2] = a[n - i - 2], a[i + 1]
