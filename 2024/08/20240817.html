It's possible to make the tower height m if the remaining moves are couples of (+1, -1).

def possible_math(n: int, m: int) -> bool:
    if n < m:
        return False
    return (n - m) % 2 == 0


def possible_greedy(n: int, m: int) -> bool:
    for i in range(1, n + 1):
        if m == i - (n - i):
            return True
    return False


for _ in range(int(input())):

        print("YES")

        print("NO")
