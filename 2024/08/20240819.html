  1. find the section d belongs to

def find_section(k: int, a: list[int], d: int) -> int:
    # binary search
    li: int = 0
    ri: int = k
    while li < ri - 1:
        mi: int = (li + ri) // 2
        if a[mi] == d:
            return mi
        elif a[mi] < d:
            li = mi
        else:
            ri = mi
    return li


for tc in range(int(input())):

        _ans: int = b[i] + (d - a[i]) * (b[i + 1] - b[i]) // (a[i + 1] - a[i])
        ans.append(_ans)

    print(" ".join([str(x) for x in ans]))
