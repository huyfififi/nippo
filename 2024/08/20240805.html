from collections import defaultdict


for _ in range(int(input())):
    n: int = int(input())
    s: list[str] = list(input())

    d: defaultdict = defaultdict(int)
    for x in s:
        d[x] += 1

    score: int = 0
    score += min(d["A"], n)
    score += min(d["B"], n)
    score += min(d["C"], n)
    score += min(d["D"], n)
    print(score)
