- LeetCode 122\. Best Time to Buy and Sell Stock II
	- 1 -> 5 is equivalent to 1 -> 3 and 3 -> 5 (greedy)
	- complicated DP

```cpp
#include <algorithm>
#include <vector>

class Solution {
public:
    int maxProfit(std::vector<int>& prices) {
        int profit = 0;
        for (int i = 1; i < prices.size(); ++i) {
            profit += std::max(0, prices[i] - prices[i - 1]);
        }
        return profit;
    }
};
```
